1. Find the total revenue (price Ã— quantity) for each item, sorted from highest to lowest.

db.sales.aggregate([
  {
    $project: {
      item: 1,
      revenue: { $multiply: ["$price", "$quantity"] }
    }
  },
  {
    $group: {
      _id: "$item",
      totalRevenue: { $sum: "$revenue" }
    }
  },
  {
    $sort: { totalRevenue: -1 }
  }
]);


2. Calculate the total quantity sold per month in 2022.
db.sales.aggregate([
  {
    $match: {
      date: {
        $gte: ISODate("2022-01-01"),
        $lt: ISODate("2023-01-01")
      }
    }
  },
  {
    $group: {
      _id: { month: { $month: "$date" } },
      totalQuantity: { $sum: "$quantity" }
    }
  },
  {
    $sort: { "_id.month": 1 }
  }
]);

10. Find all sales where the price is greater than the average price of their respective size.
db.sales.aggregate([
  {
    $group: {
      _id: "$size",
      avgPrice: { $avg: "$price" }
    }
  },
  {
    $lookup: {
      from: "sales",
      localField: "_id",
      foreignField: "size",
      as: "sales"
    }
  },
  {
    $unwind: "$sales"
  },
  {
    $match: {
      $expr: { $gt: ["$sales.price", "$avgPrice"] }
    }
  },
  {
    $replaceRoot: { newRoot: "$sales" }
  }
]);


11. Find Sales Where the Day of Week Matches Quantity's Last Digit
db.sales.aggregate([
  {
    $addFields: {
      dayOfWeek: { $dayOfWeek: "$date" },
      quantityLastDigit: {
        $mod: ["$quantity", 10]
      }
    }
  },
  {
    $match: {
      $expr: {
        $eq: ["$dayOfWeek", { $add: ["$quantityLastDigit", 1] }]
      }
    }
  }
]);
//MongoDB $dayOfWeek returns values from 1 (Sunday) to 7 (Saturday). add 1 to the digit to match this range.



12. Find Sales Where the Month is Prime and Quantity is Odd
db.sales.aggregate([
  {
    $addFields: {
      month: { $month: "$date" }
    }
  },
  {
    $match: {
      month: { $in: [2, 3, 5, 7, 11] },
      quantity: { $mod: [2, 1] }
    }
  }
]);
